-- Database: DiplomaDB

-- DROP DATABASE IF EXISTS "DiplomaDB";

CREATE DATABASE "DiplomaDB"
    WITH
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'Russian_Russia.1251'
    LC_CTYPE = 'Russian_Russia.1251'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1
    IS_TEMPLATE = False;
	
CREATE TABLE "Categories"(
	id serial PRIMARY KEY,
	title text,
	role integer NOT NULL,
	photourl text
);
	
CREATE TABLE "Tests"(
	id serial PRIMARY KEY,
	question text NOT NULL,
	testbody jsonb NOT NULL,
	testname text NOT NULL,
	categoryid integer REFERENCES "Categories"(id) NOT NULL,
	videourl text,
	imageurl text,
);
	
CREATE TABLE "Videos"(
	id serial PRIMARY KEY,
	link text,
	title text,
	categoryid integer REFERENCES "Categories"(id)
);

CREATE TABLE "Users"(
	id serial PRIMARY KEY,
	username text NOT NULL,
	email text NOT NULL UNIQUE,
	passwordhash text NOT NULL,
	role integer NOT NULL,
	verificationtoken text,
	resettoken text,
	resettokenexpires timestamp,
	progress real NOT NULL DEFAULT 0,
	groupid integer REFERENCES "Groups"(id)
);

CREATE TABLE "RefreshTokens" (
	id serial PRIMARY KEY,
	userid integer REFERENCES "Users"(id) NOT NULL,
	token text,
	expires timestamp NOT NULL,
	created timestamp NOT NULL,
	createdbyip text,
	revoked timestamp,
	revokedbyip text,
	replacedbytoken text,
	reasonrevoked text
);

CREATE TABLE "CompletedTasks"(
	id serial PRIMARY KEY,
	categoryid integer REFERENCES "Categories"(id) NOT NULL,
	userid integer REFERENCES "Users"(id) NOT NULL
);

CREATE TABLE "Groups" (
	id serial PRIMARY KEY,
	name text NOT NULL
);

CREATE TABLE "TestResults" (
	id serial PRIMARY KEY,
	userid integer REFERENCES "Users"(id) NOT NULL,
	result text NOT NULL,
	passdate timestamp NOT NULL,
	categoryid integer REFERENCES "Categories"(id) NOT NULL,
	groupid integer REFERENCES "Groups"(id)
);

CREATE TABLE "RandomTasks" (
	id serial PRIMARY KEY,
	videourl text,
	imageurl text,
	correctanswer text NOT NULL
);

CREATE TABLE "Articles" (
	id serial PRIMARY KEY,
	title text NOT NULL,
	description text NOT NULL,
	htmltext text NOT NULL,
	imageurl text NOT NULL,
	role integer NOT NULL,
	groupid integer REFERENCES "Groups"(id)
);
